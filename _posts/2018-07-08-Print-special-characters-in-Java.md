<h3>Problem</h3>

<blockquote><p>
다음과 같이 출력하시오.<br>
|\_/|<br>
|q p|   /}<br>
( 0 )"""\<br>
|"^"`    |<br>
||_/=\\__|</p></blockquote>

<h3>Code</h3>

{% highlight python %}


public class P10172 {

	public static void main(String[] args) {
		
		System.out.println("|\\_/|");
		System.out.println("|q p|   /}");
		System.out.println("( 0 )\"\"\"\\");
		System.out.println("|\"^\"`    |");
		System.out.println("||_/=\\\\__|");

	}

}
